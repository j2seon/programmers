class Solution {
    public long solution(int a, int b) {
        long answer = 0;
        
    if(a<b){
        for(int i= a; a<=b; a++){
            answer+=a;
        }
    }else if(a>b){
         for(int i= b; b<=a; b++){
            answer+=b;
        }
    }else{
        answer=a;
    }       
        return answer;
    }
    public static void main (String[] args){
        Solution sol = new Solution();
        System.out.println(sol.solution(3, 5));
    }
}
